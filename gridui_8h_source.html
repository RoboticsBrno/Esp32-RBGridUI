104 T* newWidget(float x, float y, float w, float h, uint16_t uuid, uint16_t tab) {
105 static_assert(std::is_base_of<builder::Widget, T>::value, "T must inherit from builder::Widget.");
106
107 std::lock_guard<std::mutex> l(m_states_mu);
108
109 if (!checkUuidFreeLocked(uuid))
110 uuid = generateUuidLocked();
111
112 auto* state = new WidgetState(uuid, x, y, w, h, tab, &T::callbackTrampoline);
113 m_states.push_back(std::unique_ptr<WidgetState>(state));
114
115 auto* widget = new T(T::name(), *state);
116 m_widgets.push_back(std::unique_ptr<T>(widget));
117 return widget;
118 }
